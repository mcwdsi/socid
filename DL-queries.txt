inverse 'is about' min 2 'OMB racial identity datum' or inverse 'is about' some 'PCORnet multiple race identity datum'

We can't construct a DL query that retrieves everyone who self-identifies at all, but 

inverse 'is about' some ('PCORnet multiple race identity datum' and is_specified_output_of some ('racial identification process' and 'has active participant' some (inverse 'is about' some 'PCORnet multiple race identity datum')))

This works for this synthetic data set, but will it work for aggregating data across pcornet databases? Perhaps query for the processes themselves?
But you can get a list of all of the racial identification processes that are self-identifying, and with SPARQL you can list their outputs.

SPARQL

#Retreive all of the humans
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX obo: <http://purl.obolibrary.org/obo/>

SELECT ?person
	WHERE {?person rdf:type obo:NCBITaxon_9606} #person is a Homo sapiens
	
#Which of the study participants are racially identified as Asian according to the OMB criteria? 
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX obo: <http://purl.obolibrary.org/obo/>

SELECT DISTINCT  ?person
	WHERE 
	{?person a obo:NCBITaxon_9606 . #person is a Homo sapiens
	 ?identity a obo:OMRSE_00000181 .  #identity is an Asian identity datum
	 ?identity obo:IAO_0000136 ?person #identity is about person
	}
	
#Which of the study participants are racially identified with multiple races according to OMB criteria?
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX obo: <http://purl.obolibrary.org/obo/>

SELECT DISTINCT  ?person
	WHERE 
	{
	 ?person a obo:NCBITaxon_9606 . #person is a Homo sapiens
	 ?identity a obo:OMRSE_00000185 . #identity is an OMB racial identity datum
	 ?identity obo:IAO_0000136 ?person #identity is about person
	}
	
GROUP BY ?identity
HAVING (COUNT(DISTINCT ?identity) >=2) #with a minimum of two identities

#who self-identified and what is their racial identity?
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT DISTINCT  ?person ?identity
	WHERE

	?person a obo:NCBITaxon_9606 . #person is a human being

	}
